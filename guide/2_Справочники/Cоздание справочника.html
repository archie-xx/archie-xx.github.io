# Создание нового справочника

# Backend

# Создание модели

В папке Models/AdminModels создаем новый файл, аналогичный названию справочника в един числе


use Mintrocket\mint_admin\Models\AdminModels\CRUDModel;

class Example extends CRUDModel
{
    protected $fillable = [
        'name',
    ];

    protected $hidden = [
        'created_at',
        'updated_at',
    ];
}

# Создание контроллера

В папке Controllers/AdminControllers создаем новый файл, аналогичный названию справочника тоде в един числе

  • Подключаем туда ранее созданную модель, CRUDController
  • Прописываем публичную переменную $modelClass
  • При необходимости переопределяем хуки before_* after_*
  • При необходимости переопределяем FormRequest (наследуя соотвествующие реквесты) по пути App\Http\Requests\Admin\CRUD\* реквесты DestroyRequest IndexRequest ShowRequest StoreRequest UpdateRequest
namespace App\Http\Controllers\AdminControllers;

use App\Models\Example;
use Mintrocket\mint_admin\Http\Controllers\CRUDController;

class ExampleController extends CRUDController
{
    public $modelClass = Example::class;
}

# Роутинг

В файле routes/api.php добавляем в массив $apiResources созданный контроллер и путь во множ числе по которому фронтенд будет ссылаться на бэкенд

$apiResources = [
   'examples' => 'ExampleController',
];

# Frontend

Идем в папку resources/js/models, там создаем файл, аналогичный созданной модели в Ларе, напр

# !!! В нем обязательно должны совпадать url и путь к контроллеру, который мы ранее прописали в $apiResources файлa routes/api.php иначе запросы слаться будут не туда куда нужно


//Example.js

import ORM from '../api/ORM';
import RequestLifecycle from '../api/lifecycle';

let model_name = 'examples';

let domain = process.env.MIX_API_HOST + 'api/v1/admin/';

export default {
  name: "Пример",
  url: "examples",
  icon: "",

  searchable: false,

  show: {
    add: false,
    edit: true,
    delete: false
  },

  table: {

    sort: {
      sortBy: 'id',
      sortDirection: 'desc',
    },

    fields: [{
      key: "id",
      label: "#"
    }],
  },
  form: {
    fields: [{}]
  },
  methods: {
    async all(type = null) {
      return await RequestLifecycle({ method: 'GET', url: `${domain}${model_name}?all${type ? `&select=${type}` : ''}` });
    },
    async paginate({page, perPage, query, sortBy, sortDirection, params}) {
               return await ORM.paginate(
                   model_name,
                   page,
                   perPage,
                   query,
                   sortBy,
                   sortDirection,
               );
           },,
    async find(id) {
      return await ORM.find(model_name, id);
    },
    async create(data) {
      return await ORM.create(model_name, data);
    },
    async update(id, data) {
      return await ORM.update(model_name, id, data);
    },
    async delete(id) {
      return await ORM.delete(model_name, id);
    }
  }
}

В routes/index.js импортим Examples.js


import Example from './Example';

export default {
	Example
}

Далее в resources/js/views/models/ создаем папку с названием модели в множественном числе (прям как url) в нижнем регистре

Там же помещаем файлы Form.vue && Index.vue

в Form.vue импортим нашу модель


<!-- Form.vue -->

<template>
  <div>
    <Form :model="model"/>
  </div>
</template>

<script>
import Form from "../../../components/CRUD/Form";
import model from "../../../models/Example";

export default {
  name: "Example",

  data() {
    return {
      model
    };
  }, 
  components: {
    Form
  }
};
</script>

в Index.vue также импортим модель Example.js


<template>
  <div>
    <Index :model="model" > </Index>
  </div>
</template>

<script>
import Index from "../../../components/CRUD/Index";
import model from "../../../models/Example";
export default {
  name: "UserIndex",
  data() {
    return {
      model
    };
  },
  components: {
    Index
  }
};
</script>

Теперь справочник доступен по url, который мы указывали в Example.js
