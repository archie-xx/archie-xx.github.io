# Представление редактирования

# Добавление полей на форму

//resources/js/models/Example.js
import { FILE } from "../assets/constants/form/field-types";

...
...
...

 form: {
    fields: [
        {
          label: 'Example',
          type: FILE,
          placeholder: 'Example',
          key: 'file',
          value: '',
          required: true,
        }
    ]
  },

[Типы полей](Типы полей формы.md) доступны в файле ../assets/constants/form/field-types

Кроме того, вы можете использовать обычные типы input'a. Например, такие как text, password, email и т.п.

 form: {
    fields: [{
      label: 'Example',
      type: 'password',
      placeholder: 'Example',
      key: 'password',
      value: '',
      required: true,
    }]
  },

# Настройка полей формы

# Валидация значения

Для каждого поля можно назначить правила валидации значения rules:

import { minLength, email, required } from 'vuelidate/lib/validators'
...

 form: {
    fields: [{
      label: 'Пароль',
      type: 'password',
      placeholder: 'Пароль',
      key: 'password',
      required: true,
      rules: [
        {
          validator: required,
          message: 'Обязательное поле'
        },
        {
          validator: email,
          message: 'Неправильный формат почты'
        }
      ]
    }]
  }

Для валидации используется библиотека Vuelidate. Встроенные методы валидации доступны по здесь (opens new window)

Также есть возможность использовать пользовательский валидатор:

 form: {
    fields: [{
      label: 'Пароль',
      type: 'password',
      placeholder: 'Пароль',
      key: 'password',
      rules: {
          validator: async (value, fields) => {
            await fetch('emailExists')
          },
          message: 'Данный эл. адрес уже используется'
      }
    }]
  }

Первым параметром передается значение валидируемого поля, вторым объект со значениями остальных полей формы

# Настройка видимости

Для каждого поля можно изменить видимость visible:

 form: {
    fields: [{
      label: 'Example',
      type: 'text',
      placeholder: 'Example',
      key: 'name',
      required: true,
      visible: false
    }]
  }

Для более детальной настройки необходимо прописать правило видимости visibilityRule:

 form: {
    fields: [{
      label: 'Example',
      type: 'text',
      placeholder: 'Example',
      key: 'name',
      required: true,
      visible: false,
      visibilityRule: {
        fields: ['password', 'passwordConfirm'],
        validator: (password, passwordConfirm) => password === passwordConfirm
      }
    }]
  }

В правиле нужно указать массив ключей полей, от которых зависит текущее, и валидатор его значения (validator). В параметры функции-валидатора будут переданы значения указанных в fields полей в соответствующем порядке.

Поле будет скрыто если валидатор вернет false

Функция-валидатор может быть асинхронной
