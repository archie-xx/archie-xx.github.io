# Типы полей формы

# Параметры доступные всем полям

* обязательное

  • key* - соответствующий ключ, используемый для передачи и получения данных с бэкенда
  • type* - тип поля
  • label - подпись к полю
  • value - начальное значение (должно иметь определенный тип данных в зависимости от типа поля)

# File

value: Array

  • options*: {
    • multiple: возможность загружать один или несколько файлов (true/false)
    • max_files: макс кол-во файлов
    • accept: тип доступных для загрузки файлов
    • source: в дальнейшем это поле будет использоваться как collection_name
    • key: по дефолту остается original_filename }
{
    label: 'Изображения',
    type: FILE,
    options: {
        multiple: false,
        max_files: 5,
        accept: 'image/*',
        source: 'imgs',
        key: 'original_filename',
    },
    key: 'imgs',
    value: [],
}

# Select

value: Array

  • select*: {
    • multiple: Возможность выбрать несколько полей (true/false)
    • source: Источник данных: обращаемся к модели, к ее объекту methods => методу
    • label: Из получаемого объекта в source берем ключ, который будет юзаться в селекте обычным текстом
    • value: Из того же объекта берем ключ, который будет соответствовать value
    • view: {
      • source: источник как в экземле, (???) если кто знает как это юзается, то исправьте плиз //TODO исправить
      • key == view.label } }
//select
{
    key: 'order_category_id',
    label: 'Категория',
    type: SELECT,
    select: {
        multiple: false,
        source: Category.methods.all,
        label: 'name',
        value: 'id',
        view: {
            source: 'category',
            key: 'name'
        }
    },
    value: [],
},

# Добавление события onChange

За основу имеем select, который описан выше. На первом уровне объекта мы можем объявить функцию onChange(), которая работает по аналогии JavaScript event'а onChange и принимает value (значение выбранного селекта)

Мы можем использовать это событие, например, для реализации последовательного выбора из нескольких выпадающих списков, где возможные значения следующего списка зависят от выбранного значения в предыдущем.

let subcategories = { // AHTUNG
  status: true,
  data: [],
}

// ... //
let model = {
    // ... //
    fields: [
        {
             label: 'Вид спорта',
             type: SELECT,
             select: {
                 multiple: false,
                 source: Sport.methods.all,
                 label: 'name',
                 value: 'id',
                 view: {
                     source: 'sport',
                     key: 'name'
                 }
             },
             value: null,
             key: 'sport_id',
             required: true,
             onChange: async value => {
               subcategories.data = [];
               if (value == null) {
                 return;
               }
               model.form.fields.find(field => field.key === 'subcategory_id').value = null;
               subcategories.data = (await Subcategory.methods.all({ sport_id: value.value})).data.data;
             }
         }, {
             label: 'Подкатегория',
             type: SELECT,
             select: {
                 multiple: false,
                 source: subcategories, // AHTUNG
                 label: 'name',
                 value: 'id',
                 view: {
                     source: 'subcategory',
                     key: 'name'
                 }
             },
             value: null,
             key: 'subcategory_id',
             required: true,
         }
    ]
}

export default model;
