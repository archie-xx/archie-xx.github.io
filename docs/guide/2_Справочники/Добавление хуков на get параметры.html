# Добавление хуков на get параметры

Добавление новых полей в Form:

  • let slots = переменная для хранения полей, которые будут отбражаться

  • model.defaultSlot объект, который содержит значения для полей по умолчанию.

  • model.routeGetParams - массив объектов хуков

  • model.routeGetParams.index - для хуков на index странице

  • model.routeGetParams.form - для хуков на form странице

  • model.routeGetParams.index.*.source - названия ключа в get параметрах

  • model.routeGetParams.index.*.target - (string, array, undefined) - значения поля key объекта из массива fields

  • model.routeGetParams.index.*.onUpdate - async (value, field, model) => {} асинхронная колбэк функция. принимает значение гет параметра - value, и поле field - объект инпута формы( или массив ) и объект model

//resources/js/models/Example.js
 let slots = {
   index: {
     topHeader: `текс по умолчанию`
   },
   form: {
     topHeader: `текс по умолчанию`
   }
 };


let model = {
  slots: slots,
  defaultSlots: {
    index: {
      topHeader: `текс по умолчанию`
    },
    form: {
      topHeader: `текс по умолчанию`
    }
  },
  routeGetParams: {
      index:[

    {
      source: "get_param_name",//key в get параметрах
      target: "field_key", // значение поля key у поля формы
      onUpdate: async (value, field, model) => {
          
        let q =  ORM.find(value); // доп запрос 
        field.value = {
          label: q.name,
          value: value
        };        
         
        slots.form.topHeader ='custom text for slot '+ q.name;
      }
    }
 ]

  
}
/// ...
};

Для взаимодействия пагинации и гет параметра надо добавить название гет параметра в массив, который прокидывается в ORM.paginate 7ым параметром.

{
methods: {
        async all() {
            return await ORM.all(model_name);
        },
        async paginate(page, perPage, query, sortBy, sortDirection, params) {
            return await ORM.paginate(
                model_name,
                page,
                perPage,
                query,
                sortBy,
                sortDirection,
                [ 'role_id'], // массив гет параметров, которые будут прокидоваться в метод пагинации
                params
            );
        }
//...
}
}
